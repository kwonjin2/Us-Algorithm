# [92341] 주차 요금 계산

```python
# 입차, 출차 기록이 주어졌을 때 차량별로 주차 요금을 계산하려고 한다.
# 기본 시간(분) : 180 원
# 기본 요금(원) : 5000 원
# 단위 시간(분) : 10 원
# 단위 요금(원) : 600 원
def solution(fees, records):
    answer = []
    base_hour, base_pay, unit_hour, unit_pay = fees
    
    dict = {}
    for record in records:
        number = record.split(" ")
        
        if not number[1] in dict:
            dict[number[1]] = [to_minutes(number[0])]
        else: 
            dict[number[1]].append(to_minutes(number[0]))
            
    sorted_records = sorted(dict.items(), key=lambda item:item[0], reverse=False)
    
    for car_number, car_inout in sorted_records:
        num = 0
        result = 0

        for i in range(len(car_inout)):
            if i % 2 != 0:
                num += (car_inout[i] - car_inout[i - 1])
            elif len(car_inout) % 2 != 0 and len(car_inout) - 1 == i:
                num += 1439 - car_inout[-1]
        
        extra = max(0, num - base_hour)  
        if num < base_hour:
            result = base_pay
        else:
            extra = max(0, num - base_hour)  
            result = base_pay + ((extra + unit_hour - 1) // unit_hour) * unit_pay
        answer.append(result)            
            
    return answer


def to_minutes(time):
    hour, minute = map(int, time.split(":"))
    return hour * 60 + minute
```
