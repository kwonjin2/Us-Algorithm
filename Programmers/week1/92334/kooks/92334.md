import java.util.*;

class Solution {
    public int[] solution(String[] id_list, String[] report, int k) {
        Map<String, Set<String>> reportMap = new HashMap<>(); //신고당한 유저 -> 신고한 사람들 
        Map<String, Integer> mailCount = new HashMap<>(); //메일 받은 횟수
        
        for(String id : id_list){
            reportMap.put(id, new HashSet<>());
            mailCount.put(id, 0);
        }
        
        //중복 신고 제거 및 기록
        for(String rep : report){
            String[] parts = rep.split(" ");
            String reporter = parts[0];
            String reported = parts[1];
            reportMap.get(reported).add(reporter);
        }
        
        for(String reported : id_list){
            Set<String> reporters = reportMap.get(reported);
            if(reporters.size() >= k){
                for(String reporter : reporters){
                    mailCount.put(reporter, mailCount.get(reporter) + 1);
                }
            }
        }
        
        int[] result = new int[id_list.length];
        for(int i =0; i < id_list.length; i++){
            result[i] = mailCount.get(id_list[i]);
        }
        
        return result;
    }
}

def solution(id_list, report, k):
    # 1. 중복 제거
    report = set(report)
    
    # 2. 신고 기록 분리
    reported_count = {user: 0 for user in id_list}   # 피신고자 신고 횟수
    reporter_map = {user: [] for user in id_list}    # 신고자 → 피신고자 리스트
    
    for r in report:
        reporter, reported = r.split()
        reporter_map[reporter].append(reported)
        reported_count[reported] += 1
    
    # 3. 정지 대상 찾기
    suspended = {user for user, cnt in reported_count.items() if cnt >= k}
    
    # 4. 각 유저별 메일 카운트
    answer = []
    for user in id_list:
        count = sum(1 for target in reporter_map[user] if target in suspended)
        answer.append(count)
    
    return answer
