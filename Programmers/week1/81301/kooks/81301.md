```python
def solution(s):
    answer = 0
    number_dict ={"zero":0, "one":1, "two":2, "three":3, "four":4, "five":5, "six":6, "seven":7, "eight":8, "nine":9}
    
    word = ""
    for alphaber in s:
        if alphaber.isalpha():
            word += alphaber
            if word in number_dict:
                answer = (answer * 10) + number_dict[word]
                word = ""
        else:
            answer = (answer * 10) + int(alphaber)
                
    
    return answer
```
