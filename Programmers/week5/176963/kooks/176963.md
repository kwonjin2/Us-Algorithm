# [176963] 추억 점수

```java
// 각각의 인물에 대한 추억의 점수가 있음
// 해당 사긴의 인물 점수에 따라 사진 점수가 매겨짐
// 인물 점수가 없다면 0 처리
import java.util.*;

class Solution {
    public int[] solution(String[] name, int[] yearning, String[][] photo) {
        List<Integer> answer = new ArrayList<>();
        Map<String, Integer> manScore = new HashMap<>();
        
        for(int i =0; i< name.length; i++){
            manScore.put(name[i], yearning[i]);
        }
        
        for(String[] photos : photo){
            int sum = 0;
            for(String man : photos){
                if(manScore.containsKey(man)){
                    sum += manScore.get(man);    
                }
            }
            answer.add(sum);
        }
        return answer.stream().mapToInt(Integer::intValue).toArray();
    }    
        
}
```
