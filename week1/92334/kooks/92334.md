def solution(id_list, report, k):
    # 1. 중복 제거
    report = set(report)
    
    # 2. 신고 기록 분리
    reported_count = {user: 0 for user in id_list}   # 피신고자 신고 횟수
    reporter_map = {user: [] for user in id_list}    # 신고자 → 피신고자 리스트
    
    for r in report:
        reporter, reported = r.split()
        reporter_map[reporter].append(reported)
        reported_count[reported] += 1
    
    # 3. 정지 대상 찾기
    suspended = {user for user, cnt in reported_count.items() if cnt >= k}
    
    # 4. 각 유저별 메일 카운트
    answer = []
    for user in id_list:
        count = sum(1 for target in reporter_map[user] if target in suspended)
        answer.append(count)
    
    return answer
